94 void load_packed_protacker_music(char *filename)
95 {
96  int unpacked_block_size, packed_block_size;
97  BPTR fileHandle;
98  char tag[4];
99  UWORD mod_size;
100  BYTE *packed_block, *unpacked_block;
101 
102  if ((fileHandle = Open(filename, MODE_OLDFILE)))
103  {
104  Read(fileHandle, &tag, 4);
105  if (strncmp(tag, "PTPK", 4) == 0)
106  {
107  // read original module size
108  Read(fileHandle, &unpacked_block_size, 4);
109  unpacked_block = AllocMem(unpacked_block_size, MEMF_CHIP);
110 
111  Read(fileHandle, &tag, 4);
112  if (strncmp(tag, "MINZ", 4) == 0)
113  {
114  Read(fileHandle, &tag, 4);
115  if (strncmp(tag, "SIZE", 4) == 0)
116  {
117  Read(fileHandle, &packed_block_size, 4);
118  packed_block = AllocMem(packed_block_size, 0L);
119  Read(fileHandle, packed_block, packed_block_size);
120 #ifdef DEBUG_MACROS
121  printf("!!!!MINIZ block size: %d\n", packed_block_size);
122 #endif
123  tinfl_decompress_mem_to_mem(unpacked_block, unpacked_block_size, packed_block, packed_block_size, 1);
124  FreeMem(packed_block, packed_block_size);
125  }
126  else
127  printf(", no 'SIZE' tag found!");
128  }
129  else if (strncmp(tag, "SHRK", 4) == 0)
130  {
131  Read(fileHandle, &tag, 4);
132  if (strncmp(tag, "SIZE", 4) == 0)
133  {
134  Read(fileHandle, &packed_block_size, 4);
135  packed_block = AllocMem(packed_block_size, 0L);
136  Read(fileHandle, packed_block, packed_block_size);
137 #ifdef DEBUG_MACROS
138  printf("!!!!SHRK block size: %d\n", packed_block_size);
139 #endif
140  ShrinklerDecompress(packed_block, unpacked_block, NULL, NULL);
141  FreeMem(packed_block, packed_block_size);
142  }
143  else
144  printf(", no 'SIZE' tag found!");
145  }
146  else if (strncmp(tag, "D68K", 4) == 0)
147  {
148  Read(fileHandle, &tag, 4);
149  if (strncmp(tag, "SIZE", 4) == 0)
150  {
151  Read(fileHandle, &packed_block_size, 4);
152  packed_block = AllocMem(packed_block_size, 0L);
153  Read(fileHandle, packed_block, packed_block_size);
154 #ifdef DEBUG_MACROS
155  printf("!!!!D68K block size: %d\n", packed_block_size);
156 #endif
157  doynaxdepack(packed_block, unpacked_block);
158  FreeMem(packed_block, packed_block_size);
159  }
160  else
161  printf(", no 'SIZE' tag found!");
162  }
163  else
164  {
165  printf("!Unknown tag: %s\n", tag);
166  FreeMem(unpacked_block, unpacked_block_size);
167  unpacked_block = NULL;
168  unpacked_block_size = 0;
169  }
170 
171  }
172  else
173  {
174  printf("!Not a Protracker Packed File!\n");
175  }
176 
177  }
178 
179  protracker_mod_data = unpacked_block;
180  protracker_mod_size = unpacked_block_size;
181 }

202 void play_protracker_music(void)

206  {

210  PTInstallBits(protracker_mod_playing, protracker_SigBit, -1, -1, -1);
211  PTSetVolume(protracker_mod_playing, protracker_mod_volume);
212  PTPlay(protracker_mod_playing);
213  }
214  else
215  printf("play_protracker_music() Couldn't allocate signal!\n");
216 
217  }
218  else
219  rpage_system_alert("Ptreplay.library was not open!");
220 }
